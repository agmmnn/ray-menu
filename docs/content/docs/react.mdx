# React Bindings

React bindings that wrap the `<ray-menu>` Web Component are planned for a future release.

In the meantime, you can use the Web Component directly in React:

```tsx
import { useEffect, useRef } from 'react';
import 'ray-menu';

function App() {
  const menuRef = useRef<HTMLElement | null>(null);

  useEffect(() => {
    const el = document.createElement('ray-menu') as any;
    el.items = [
      { id: 'copy', label: 'Copy', shortcut: '⌘C' },
      { id: 'paste', label: 'Paste', shortcut: '⌘V' },
      { id: 'cut', label: 'Cut', shortcut: '⌘X' },
    ];
    document.body.appendChild(el);
    menuRef.current = el;

    el.addEventListener('ray-select', (e: CustomEvent) => {
      console.log('Selected:', e.detail.label);
    });

    return () => el.remove();
  }, []);

  return (
    <div
      onContextMenu={(e) => {
        e.preventDefault();
        (menuRef.current as any)?.open(e.clientX, e.clientY);
      }}
    >
      Right-click here
    </div>
  );
}
```
